for (let i = 0; i < 10000; i++) {
    db.partidos.insertOne({
        sede: "Sede_" + (i % 10),
        categoria: "Categoria_" + (i % 5),
        fecha: new Date(),
        equipo1: "Equipo_A_" + i,
        equipo2: "Equipo_B_" + i
    });
}
